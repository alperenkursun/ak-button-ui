## Install

```bash
npm install --save ak-button-ui
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'ak-button-ui'
import 'ak-button-ui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
